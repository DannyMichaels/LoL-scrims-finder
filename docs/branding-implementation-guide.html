1. How It Works

User visits app.example.com | React app loads (shows dark loading screen) | GET /api/branding?hostname=app.example.com | Server matches hostname → returns brand config | createBrandTheme(config.colors) builds UI theme | Updates favicon, meta tags, manifest | Renders with <ThemeProvider> + <BrandingContext>

Core idea: The database stores a branding config per tenant, each with a list of domains. The frontend sends window.location.hostname to the API, gets back the matching config, and uses it to build the theme and inject brand text/assets.

Hostname lookup logic

async function getBrandingByHostname(req, res) {
  const { hostname } = req.query;

  // 1. Check cache
  const cached = cache.get(hostname);
  if (cached) return res.json(cached);

  // 2. Look up by domain
  let config = await BrandConfig.findOne({ domains: hostname });

  // 3. Fall back to default
  if (!config) config = await BrandConfig.findOne({ isDefault: true });

  if (!config) return res.status(404).json({ error: 'No config found' });

  cache.set(hostname, config);
  res.json(config);
}
Tip: Invalidate cache on every POST/PATCH to branding endpoints.

4. Caching

class SimpleCache {
  constructor(ttlMs = 5 * 60 * 1000) {
    this.cache = new Map();
    this.ttl = ttlMs;
  }
  get(key) {
    const entry = this.cache.get(key);
    if (!entry) return null;
    if (Date.now() - entry.ts > this.ttl) {
      this.cache.delete(key);
      return null;
    }
    return entry.value;
  }
  set(key, value) {
    this.cache.set(key, { value, ts: Date.now() });
  }
  invalidate() {
    this.cache.clear();
  }
}

For multi-server deployments, use Redis instead of in-memory.

const corsOptions = {
  origin: async function (origin, callback) {
    if (!origin) return callback(null, true);                    // server-to-server
    if (process.env.NODE_ENV === 'development') return callback(null, true);

    // Static allowlist
    if (staticOrigins.includes(origin)) return callback(null, true);

    // Dynamic: check tenant domains from DB (cached)
    const domains = await getAllDomains();
    const originHost = new URL(origin).hostname;
    if (domains.includes(originHost)) return callback(null, true);

    callback(new Error('Not allowed by CORS'));
  }
};

6. Theme Factory

MUI Example

import { createTheme, alpha, lighten, darken } from '@mui/material/styles';

function getContrastText(hex) {
  const r = parseInt(hex.substring(1, 3), 16);
  const g = parseInt(hex.substring(3, 5), 16);
  const b = parseInt(hex.substring(5, 7), 16);
  const luminance = (0.299 * r + 0.587 * g + 0.114 * b) / 255;
  return luminance > 0.5 ? '#000000' : '#FFFFFF';
}

export function createBrandTheme(colors = {}) {
  const primary = colors.primaryMain || '#2196F3';
  const primaryLight = colors.primaryLight || lighten(primary, 0.2);
  const primaryDark = colors.primaryDark || darken(primary, 0.2);

  return createTheme({
    palette: {
      mode: 'dark',
      primary: {
        main: primary, light: primaryLight, dark: primaryDark,
        contrastText: getContrastText(primary),
      },
      background: {
        default: colors.backgroundDefault || '#0a0e1a',
        paper: colors.backgroundPaper || '#121826',
      },
    },
    components: {
      // Replace ALL hardcoded colors with alpha(primary, ...) etc.
    },
  });
}

CSS Variables Alternative (non-MUI)

function applyBrandTheme(colors) {
  const root = document.documentElement;
  root.style.setProperty('--color-primary', colors.primaryMain);
  root.style.setProperty('--color-primary-light', colors.primaryLight);
  root.style.setProperty('--color-bg', colors.backgroundDefault);
}

Then use var(--color-primary) everywhere in your CSS.

Key rule: Never hardcode color hex values in component files. Always reference the theme or CSS variables.

7. Branding Provider (Loading Gate)

export const BrandingContext = createContext(null);

export default function BrandingProvider({ children }) {
  const [config, setConfig] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    api.get(`/branding?hostname=${window.location.hostname}`)
      .then(({ data }) => {
        setConfig(data);
        // Update favicon, apple-touch-icon, og:image, manifest link
      })
      .catch(console.error)
      .finally(() => setLoading(false));
  }, []);

  const theme = useMemo(() => createBrandTheme(config?.colors || {}), [config]);
  const branding = useMemo(() => ({
    brandName: config?.branding?.brandName || 'My App',
    logoUrl: config?.branding?.logoUrl || '/logo.png',
    // ... other fields
  }), [config]);

  if (loading) return <LoadingScreen />;

  return (
    <BrandingContext.Provider value={branding}>
      <ThemeProvider theme={theme}>{children}</ThemeProvider>
    </BrandingContext.Provider>
  );
}

// index.js — replace ThemeProvider with BrandingProvider

</BrandingProvider>

8. Consuming Branding Data

// hooks/useBranding.js
export default function useBranding() {
  const ctx = useContext(BrandingContext);
  if (!ctx) throw new Error('useBranding must be inside BrandingProvider');
  return ctx;
}

// In any component:
function Navbar() {
  const { brandName, logoUrl } = useBranding();
  return (
    <AppBar>
      <img src={logoUrl} alt="logo" />
      <Typography>{brandName}</Typography>
    </AppBar>
  );
}

What to find & replace

  • Hardcoded brand names → useBranding().brandName
  • Hardcoded logo paths → useBranding().logoUrl
  • Hardcoded social links → useBranding().socialLinks
  • Hardcoded hex colors → theme.palette.primary.main / alpha(...)

9. Dynamic Meta Tags

index.html — neutral defaults

<title>Loading...</title>
<link rel="icon" href="%PUBLIC_URL%/default-favicon.png" />
<meta name="description" content="Loading..." />

React Helmet for dynamic title/description

const { brandName, tagline } = useBranding();
<Helmet>
  <title>{brandName}</title>
  <meta name="description" content={`${brandName} - ${tagline}`} />
</Helmet>
Important — og:image and link previews: Client-side JS updates to <meta property="og:image"> do NOT work for social media previews (Discord, Twitter, Slack). These crawlers read raw HTML, not JavaScript output.

Solutions for social previews

  1. SSR — Next.js / Remix render <head> tags server-side per hostname
  2. Edge middleware — Cloudflare Workers, Vercel Edge Functions, or Lambda@Edge intercept HTML and inject correct <meta> tags based on hostname
  3. Prerender service — prerender.io serves bot user agents pre-rendered HTML
  4. Reverse proxy — Nginx/Apache rewrites meta tags based on Host header

Edge function example

// Cloudflare Worker / Vercel Edge Middleware
export default async function middleware(request) {
  const response = await fetch(request);
  const hostname = new URL(request.url).hostname;
  const config = await fetch(`https://api.yourapp.com/branding?hostname=${hostname}`)
    .then(r => r.json());

  return new HTMLRewriter()
    .on('meta[property="og:image"]', {
      element(el) { el.setAttribute('content', config.branding.logoUrl); }
    })
    .on('title', {
      element(el) { el.setInnerContent(config.branding.brandName); }
    })
    .transform(response);
}

Lookup query

-- By hostname
SELECT bc.* FROM brand_configs bc
JOIN brand_config_domains bcd ON bc.id = bcd.brand_config_id
WHERE bcd.domain = $1 LIMIT 1;

-- Fallback
SELECT * FROM brand_configs WHERE is_default = TRUE LIMIT 1;

-- All domains (for CORS)
SELECT DISTINCT domain FROM brand_config_domains;

Express controller with SQL

async function getBrandingByHostname(req, res) {
  const { hostname } = req.query;
  const cached = cache.get(hostname);
  if (cached) return res.json(cached);

  let result = await pool.query(`
    SELECT bc.*, array_agg(bcd.domain) as domains
    FROM brand_configs bc
    JOIN brand_config_domains bcd ON bc.id = bcd.brand_config_id
    WHERE bcd.domain = $1
    GROUP BY bc.id LIMIT 1
  `, [hostname]);

  if (result.rows.length === 0) {
    result = await pool.query(`
      SELECT bc.*, array_agg(bcd.domain) as domains
      FROM brand_configs bc
      JOIN brand_config_domains bcd ON bc.id = bcd.brand_config_id
      WHERE bc.is_default = TRUE
      GROUP BY bc.id LIMIT 1
    `);
  }

  // Reshape flat row into nested JSON for frontend compatibility
  const row = result.rows[0];
  const config = {
    _id: row.id,
    name: row.name,
    slug: row.slug,
    domains: row.domains,
    branding: {
      brandName: row.brand_name,
      tagline: row.tagline,
      logoUrl: row.logo_url,
      faviconUrl: row.favicon_url,
    },
    colors: {
      primaryMain: row.primary_main,
      primaryLight: row.primary_light,
      primaryDark: row.primary_dark,
      backgroundDefault: row.background_default,
      backgroundPaper: row.background_paper,
    },
    socialLinks: {
      discord: row.social_discord,
      twitter: row.social_twitter,
    },
  };

  cache.set(hostname, config);
  res.json(config);
}

14. Checklist

Backend

  • Create brand config model/table with domains, colors, branding fields
  • Create GET /api/branding?hostname= with domain lookup + default fallback
  • Create GET /api/branding/manifest for dynamic PWA manifest
  • Create admin CRUD endpoints (list all, create, update)
  • Add in-memory or Redis cache with TTL + invalidation on writes
  • Update CORS to dynamically allow tenant domains
  • Create seed script for default tenant

Frontend

  • Convert static theme to factory function
  • Add getContrastText() for dynamic button text color
  • Replace ALL hardcoded color hex values with theme references
  • Create BrandingProvider with loading gate
  • Create useBranding() hook
  • Wrap app in BrandingProvider
  • Replace all hardcoded brand names/logos
  • Neutralize index.html defaults
  • Dynamic title/description via React Helmet
  • Update favicon/apple-touch-icon/og:image in provider
  • (Optional) Build admin UI with live preview
  • (Optional) SSR/edge function for og:image social previews
